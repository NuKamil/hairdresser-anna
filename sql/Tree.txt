1. Templates Section
- Templates to miejsce, gdzie definiowane są szablony konceptów. Szablony te opisują, jakie encje i atrybuty są wymagane dla jednostek IFC.
- ConceptTemplate definiuje reguły i szczegóły dotyczące konkretnych klas IFC.

    * ConceptTemplate:
        Definiuje zasady dla określonej klasy IFC (np. IfcWall).
        Posiada atrybuty takie jak uuid, name, applicableSchema i applicableEntity, które określają unikalny identyfikator, nazwę oraz do jakiej jednostki IFC szablon jest stosowany.
    
    * Definitions:
        Zawiera objaśnienia dla szablonu. Może mieć dodatkowy tekst w formacie HTML.

    * Rules:
        Definiuje zestaw zasad (AttributeRule) dla atrybutów danej encji IFC, np. sprawdza, czy IfcWall ma określony typ materiału (IfcRelAssociatesMaterial).

    * SubTemplates:
        Opcjonalnie zawiera dodatkowe podszablony, które mogą rozszerzać główny szablon. Każdy podszablon może również zawierać definicje, reguły i dalsze zagnieżdżenia.


Templates
└── ConceptTemplate
    ├── Definitions
    ├── Rules
    │   └── AttributeRule
    │       └── EntityRules
    │           └── EntityRule
    ├── SubTemplates
    │   └── ConceptTemplate
    │       ├── Definitions
    │       ├── Rules
    │       │   └── AttributeRule
    │       └── SubTemplates (opcjonalnie)
    └── ApplicableSchema (atrybut)
    └── ApplicableEntity (atrybut)



Views
└── ModelView
    ├── ExchangeRequirements
    │   └── ExchangeRequirement
    ├── Roots
    │   └── ConceptRoot
    │       ├── Concepts
    │       │   └── Concept
    │       │       ├── Template (ref)
    │       │       ├── Requirements
    │       │       └── TemplateRules
    └── ApplicableSchema (atrybut)

WITH XMLNAMESPACES (
    'http://buildingsmart-tech.org/mvd/XML/1.1' AS mvd,
    'http://www.w3.org/2001/XMLSchema-instance' AS xsi
)
-- Wyodrębnianie Specyfication (ConceptRoot i Concept)
, MVDSpecification AS (
    SELECT
        ConceptRoot.value('@uuid', 'NVARCHAR(50)') AS ConceptUUID,
        ConceptRoot.value('@name', 'NVARCHAR(100)') AS ConceptName,
        ConceptRoot.value('@applicableRootEntity', 'NVARCHAR(100)') AS ApplicableEntity
    FROM @mvdXML.nodes('/mvdXML/Views/mvd:ModelView/mvd:Roots/mvd:ConceptRoot') AS ConceptRoot(ConceptRoot)
)
-- Wyodrębnianie Applicability (AttributeRules i EntityRules) powiązanych z ConceptTemplate
, MVDApplicability AS (
    SELECT
        Templates.value('@uuid', 'NVARCHAR(50)') AS TemplateUUID,
        AttrRule.value('@AttributeName', 'NVARCHAR(100)') AS AttributeName,
        EntityRule.value('@EntityName', 'NVARCHAR(100)') AS EntityName
    FROM @mvdXML.nodes('/mvdXML/Templates/mvd:ConceptTemplate') AS Templates(Templates)
    CROSS APPLY Templates.nodes('mvd:Rules/mvd:AttributeRule') AS Rules(AttrRule)
    CROSS APPLY AttrRule.nodes('mvd:EntityRules/mvd:EntityRule') AS EntityRules(EntityRule)
)
-- Wyodrębnianie Requirements powiązanych z ConceptRoot i Concept
, MVDRequirements AS (
    SELECT
        ConceptRoot.value('@uuid', 'NVARCHAR(50)') AS ConceptUUID,
        Req.value('@requirement', 'NVARCHAR(100)') AS Requirement,
        Req.value('@applicability', 'NVARCHAR(100)') AS Applicability
    FROM @mvdXML.nodes('/mvdXML/Views/mvd:ModelView/mvd:Roots/mvd:ConceptRoot/mvd:Concept/mvd:Requirements/mvd:Requirement') AS ConceptRoot(ConceptRoot)
    CROSS APPLY ConceptRoot.nodes('mvd:Requirements/mvd:Requirement') AS Requirement(Req)
)
-- Łączenie wszystkich danych w jedno zapytanie
SELECT 
    S.ConceptUUID AS SpecificationID,
    S.ConceptName AS SpecificationName,
    S.ApplicableEntity AS Entity,
    A.AttributeName,
    A.EntityName,
    R.Requirement,
    R.Applicability
FROM MVDSpecification S
LEFT JOIN MVDApplicability A ON S.ConceptUUID = A.TemplateUUID
LEFT JOIN MVDRequirements R ON S.ConceptUUID = R.ConceptUUID;
